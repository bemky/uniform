Optionally, use js to animate toggle. 

// Javascript
$(window).on('expand-toggle', function (e){
    var target = $(e.target).find('.uniformCard-collapse');
    target.toggle(target.parents('.expanded').length == 0);
    target.slideToggle(100);
})
